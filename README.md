ma <- array(1:6, c(3,2))

#> [1] 6
nrow(ma)         # size(ma, 1)
#> [1] 3

#> [1] 2
length(dim(ma))  # rank ma
#> [1] 2

#> [2,]    2    5

library(inline)
code <- "
      real, dimension(3,2) :: a
      print *, 'size of a: ', size(a)
      print *, 'size of 1 dim a : ', size(a,1)
      print *, 'size of 2 dim a : ',size(a,2)
      print *, 'rank a: ', rank(a)
      print *, 'shape a: ', shape(a)

cubefn <- cfunction(body = code, convention=".Fortran")
